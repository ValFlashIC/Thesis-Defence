\defbeamertemplate{section in toc}{sections numbered with box}{%
  	\leavevmode%
	\begin{enumbox}{\maincolor}%
		\bfseries\inserttocsectionnumber.%
	\end{enumbox}%
	\begin{tocsecbox}{\maincolor}%
	\end{tocsecbox}%
	\ \inserttocsection\par%
}

\defbeamertemplate{subsection in toc}{subsections with mybullets}{
  	\leavevmode%
	\qquad$\bullet$\ %
	\inserttocsubsection\par%
}

\setbeamertemplate{section in toc}[sections numbered with box]
\setbeamertemplate{subsection in toc}[subsections with mybullets]
