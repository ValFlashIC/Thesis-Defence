\begin{frame}
    \frametitle{Les noyaux de caclul (Kernels)}

    \textbf{\textcolor{Red}{Où appairaissent les indirections ?}}

    \bigskip

    Les matrices creuses sont utilisées dans de \textbf{nombreux kernels} :

    \point[][2][Red]{\parbox[t]{1cm}{$1\mathcal{D}$}} Multiplication \textbf{matrice} creuse-\textbf{vecteur} : SpMV

    \point[][2][Red]{\parbox[t]{1cm}{$2\mathcal{D}$}} Multiplication \textbf{matrice}-\textbf{matrice} : SpMSpM, SpGEMM, SpMM
    %
    \makebox(0,0){\put(0,2.8\normalbaselineskip){%
        $\left.\rule{0pt}{1.1\normalbaselineskip}\right\}$

    }}
    
    \point[][2][Red]{\parbox[t]{1cm}{$n\mathcal{D}$}} Les \textbf{tenseurs} en général

    \bigskip

    Chaque kernel peut être implémenté suivant \textbf{plusieurs algorithmes} :

    \point[][2][Red]{\parbox[t]{4.5cm}{\textbf{\textcolor{Red}{Inner-Product}} (IP)}} ($\geqslant1\mathcal{D}$)

    \point[][2][Red]{\parbox[t]{4.5cm}{\textbf{\textcolor{Red}{Outer-Product}} (OP)}} ($\geqslant1\mathcal{D}$)

    \point[][2][Red]{\parbox[t]{4.5cm}{\textbf{\textcolor{Red}{Gustavson}} (ou Row-wise)}} ($\geqslant2\mathcal{D}$)

\begin{frame}
    \frametitle{La hiérarchie mémoire}

    \textcolor{Red}{\textbf{Pourquoi les accès \og aléatoires \fg{} sont un problème ?}}

    \bigskip

    \parbox[t]{5cm}{Avec une matrice dense} \arrow[]{localités spatiale et temporelle}

    \parbox[t]{5cm}{Avec une \textbf{matrice creuse}} \arrow[\bfseries]{pas de localité}

    \bigskip

    Dans une architecture classique

    \point[][2][Red]{Les hiérarchies mémoires prennent avantage des accès temporellement et spatialement locals}

    \bigskip

    \textcolor{Red}{Caches} \arrow[\color{Red}]{\textbf{non-adaptés} aux accès \og aléatoires \fg{}}

    \arrow[][2][Red]{Cache remplie de zéros (lignes de cache remplies à 40\% d'éléments non-nuls)}

    \arrow[][2][Red]{Nombre de hits bas}

    \arrow[][2][Red]{\'Evincements non-nécessaires} \arrow[]{Beaucoup de \textbf{transactions mémoires} avec des \textbf{temps de réponse élevés}}

    \medskip

    \textcolor{Red}{Simple préfetcher linéaire} \arrow[][0][Red]{pas suffisant pour s'occuper du goulot d'étranglement mémoire}

\end{frame}

\begin{frame}
    \frametitle{Résumé}

    \textbf{Inner-Product} \arrow[]{accès \og aléatoires \fg{} sur l'entrée ($b$ ou $B$)} \arrow[]{accès = lecture}

    \arrow[][2][Red]{Problème : comment \textbf{rassembler} efficacement les données} \arrow[\bfseries\color{Red}]{gather}

    \arrow[\bfseries\color{Red}][4]{Intersections} des éléments non-nuls de $A$ avec ceux de $b$ ou $B$ \hfill ($1\mathcal{D}$ ou $2\mathcal{D}$)

    \bigskip

    \textbf{Outer-Product} \arrow[]{accès \og aléatoires \fg{} sur la sortie ($c$ ou $C$)} \arrow[]{accès = lecture + accumulation + écriture}

    \arrow[][2][Red]{Problème : comment \textbf{distribuer} efficacement les données} \arrow[\bfseries\color{Red}]{scatter}

    \arrow[\bfseries\color{Red}][4]{Réductions} des sommes partielles non-nuls dans $c$ ou $C$ \hfill ($1\mathcal{D}$ ou $2\mathcal{D}$)

    \bigskip

    \textbf{Gustavson} \arrow[]{accès \og aléatoires \fg{} sur l'entrée et la sortie ($B$ et $C$)}

    \arrow[][2][Red]{Les deux broblèmes de \textbf{\textcolor{Red}{gather}} et \textbf{\textcolor{Red}{scatter}} à la fois}

    \arrow[\bfseries\color{Red}][4]{Intersections} avec les \textbf{lignes de \textit{B}} seulement \hfill ($1\mathcal{D}$)

    \arrow[\bfseries\color{Red}][4]{Réductions} \og aléatoires \fg{} dans les \textbf{lignes de \textit{C}} seulement \hfill ($1\mathcal{D}$)

\end{frame}
