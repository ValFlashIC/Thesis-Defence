\begin{frame}
    \frametitle{La hiérarchie mémoire}

    \textcolor{Red}{\textbf{Pourquoi les accès dit \og aléatoires \fg{} (ou irréguliers) sont un problème ?}}

    \bigskip

    \parbox[t]{7cm}{Noyau de calcul avec une matrice dense} \arrow[]{localités spatiale et temporelle}

    \parbox[t]{7cm}{Noyau de calcul avec une \textbf{matrice creuse}} \arrow[\bfseries]{pas de localité}

    \bigskip

    Dans une architecture classique

    \point[][2][Red]{Les hiérarchies mémoires prennent avantage des accès temporellement et spatialement locaux}

    \bigskip\bigskip

    \textbf{Caches} \arrow[\color{Red}]{\textbf{non-adaptés}} aux accès \og aléatoires \fg{}

    \arrow[][2][Red]{Cache rempli de zéros} \hfill \textcolor{black!50}{(lignes de cache remplis à 40\% d'éléments non-nuls)}

    \arrow[][2][Red]{Nombre de \textit{hits} bas}

    \arrow[][2][Red]{\'Evincements non-nécessaires}
    
    \arrow[][6]{Beaucoup de \textbf{transactions mémoires} avec des \textbf{temps de réponse élevés}}

    \medskip

    \textbf{Simple \textit{prefetcher} linéaire} \arrow[\bfseries\color{Red}]{pas suffisant} pour s'occuper du goulot d'étranglement mémoire

\end{frame}

\begin{frame}
    \frametitle{Les noyaux de calcul (Kernels)}

    Les matrices creuses sont utilisées dans de \textbf{nombreux \textit{kernels}} :

    \point[][2][Red]{\parbox[t]{1cm}{$1\mathcal{D}$}} Multiplication \textbf{\textcolor{Red}{Matrice}} creuse-\textbf{\textcolor{Red}{Vecteur}} : SpMV

    \point[][2][Red]{\parbox[t]{1cm}{$2\mathcal{D}$}} Multiplication \textbf{\textcolor{Red}{Matrice}}-\textbf{\textcolor{Red}{Matrice}} : SpMSpM, SpGEMM, SpMM
    %
    \makebox(0,0){\put(0,2.8\normalbaselineskip){%

    \point[][2][Red]{\parbox[t]{1cm}{$n\mathcal{D}$}} Les \textbf{tenseurs} en général

    \bigskip\bigskip

    \textbf{\textcolor{Red}{SpMM, SpMSpM :}} $A \times B = C$

    \medskip

    \point[][2][Red]{\underbar{Inner-Product :}} Indirections sur toute la matrice d'\textbf{entrée} $B$

    \medskip

    \point[][2][Red]{\underbar{Outer-Product :}} Indirections sur toute la matrice de \textbf{sortie} $C$

    \medskip

    \point[][2][Red]{\underbar{Gustavson :}} Indirections sur les lignes de $B$ et dans les lignes de $C$

    \point[\bfseries][6]{Intermédiaire} entre Inner-Product et Outer-Product

\begin{frame}
    \frametitle{Résumé}

    \pause[1]

    \textbf{Inner-Product (IP)} \arrow[]{accès \og aléatoires \fg{} sur l'entrée ($b$ ou $B$)}
    
    \textbf{\textcolor{white}{Inner-Product (IP)}} \arrow[]{accès = lecture}

    \arrow[][2][Red]{Problème : comment \textbf{rassembler} efficacement les données} \arrow[\bfseries\color{Red}]{gather}

    \arrow[\bfseries\color{Red}][4]{Intersections} des éléments non-nuls de $A$ avec ceux de $b$ ou $B$ \hfill ($1\mathcal{D}$ ou $2\mathcal{D}$)

    \medskip

    \pause[2]

    \textbf{Outer-Product (OP)} \arrow[]{accès \og aléatoires \fg{} sur la sortie ($c$ ou $C$)}

    \textbf{\textcolor{white}{Outer-Product (OP)}} \arrow[]{accès = lecture + accumulation + écriture}

    \arrow[][2][Red]{Problème : comment \textbf{distribuer} efficacement les données} \arrow[\bfseries\color{Red}]{scatter}

    \arrow[\bfseries\color{Red}][4]{Réductions} des sommes partielles non-nulles dans $c$ ou $C$ \hfill ($1\mathcal{D}$ ou $2\mathcal{D}$)

    \medskip

    \pause[3]

    \textbf{Gustavson (Gust)} \arrow[]{accès \og aléatoires \fg{} sur l'entrée et la sortie ($B$ et $C$)}

    \arrow[][2][Red]{Les deux problèmes de \textbf{\textcolor{Red}{gather}} et \textbf{\textcolor{Red}{scatter}} à la fois}

    \arrow[\bfseries\color{Red}][4]{Intersections} avec les \textbf{lignes de \textit{B}} seulement \hfill ($1\mathcal{D}$)

    \arrow[\bfseries\color{Red}][4]{Réductions} \og aléatoires \fg{} dans les \textbf{lignes de \textit{C}} seulement \hfill ($1\mathcal{D}$)

    \bigskip
