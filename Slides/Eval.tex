\section[Evaluations]{Evaluations et performances de SpDCache}

\begin{frame}
    \frametitle{Environnement pour l'évaluation}

    \textbf{\textcolor{Red}{Environnement RTL}} (suite à un modèle Python)

    \point[\bfseries][2][Red]{\parbox[t]{4cm}{C\oe{}ur :}} CVA6 (RISC-V) -- Scalaire, exécution dans l'ordre

    \point[\bfseries][2][Red]{\parbox[t]{4cm}{Cache L1 :}} SpDCache (\textit{mode réduction} activable à l'exécution)

    \point[\bfseries][2][Red]{\parbox[t]{4cm}{Mémoire principale :}} RAM à latence fixe de 64 cycles

    \smallskip

    \textbf{\textcolor{Red}{Base RTL :}} HPDCache~\cref{ref:HPDcache}, cache de données haute performance, libre accès (OpenHW Group)

    \point[][2][Red]{Ajout de logique de contrôle, d'accumulateurs, et de la logique des régions}

    \bigskip

    \textbf{\textcolor{Red}{Approche}}

    \point[][2][Red]{SpMV en Outer-Product}

    \point[\bfseries][2][Red]{62 matrices} -- Dimensions et structures variables

    \point[][2][Red]{Configuration des régions :} \textbf{25/50/25\%} \hfill \textcolor{black!60}{(GRC/DRC/SRT, \% de la taille totale du cache)}

    \point[][2][Red]{Comparaison avec un cache générique (GC) et un cache de réduction (RedC)}

    \bigskip

    \textbf{\textcolor{Red}{Performances}}

    \point[\bfseries][2][Red]{Traffic mémoire} entre SpDCache et mémoire principale

    \point[\bfseries][2][Red]{Accélération} en fonction de la structure à grain fin des matrices

    \vspace*{12pt}
    \dref{ref:HPDcache}{%
    César Fuguet. 2023. HPDcache: Open-Source High-Performance L1 Data Cache for RISC-V Cores. In Proceedings of the 20th ACM International Conference on Computing Frontiers (CF '23). Association for Computing Machinery, New York, NY, USA, 377-378.}

\end{frame}
