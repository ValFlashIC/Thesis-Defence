    \bigskip

    \arrow[\bfseries][0][Red]{Entrées du modèle : \textcolor{Red}{génériques}}

    \point[][2][Red]{\underbar{Matrice :} dimension $M$, taille de la bande \textbf{\textit{\textcolor{Red}{w\textsubscript{B}}}}, densité en-bande \textbf{\textit{\textcolor{Red}{d\textsubscript{IB}}}}, densité hors-bande \textbf{\textit{\textcolor{Red}{d\textsubscript{OB}}}}}

    \point[][2][Red]{\underbar{Cache :} nombre d'ensembles (\textit{sets}) \textbf{\textit{\textcolor{Red}{S}}}, nombre de \textit{ways} \textbf{\textit{\textcolor{Red}{W}}}, taille d'une ligne de cache \textbf{\textit{\textcolor{Red}{l}}}}

    \subpt[][4][Red]{Taille du cache : \textbf{\textit{\textcolor{Red}{s\textsubscript{C}}}} $= S \times W \times l$}

    \medskip

    \arrow[\bfseries][0][Red]{Sortie du modèle :} estimation semi-probabiliste/simulée du \textbf{\textcolor{Red}{traffic mémoire}} du vecteur $c$
