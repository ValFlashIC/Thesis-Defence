\begin{frame}
    \frametitle{Réductions}

    \textbf{\textcolor{Red}{Qu'est-ce qu'une réduction ?}}

    \bigskip

    Une réduction sur le vecteur $c$ : $c_{indice} \acc valeur$

    \point[][2][Red]{Une \textbf{lecture} en mémoire de $c_{indice}$}

    \point[][2][Red]{Une \textbf{accumulation} pour mettre à jour $c_{indice}$ avec $valeur$}

    \point[][2][Red]{Une \textbf{écriture} en mémoire de $c_{indice}$ mis à jour}

    \medskip

    Cette opération est noté \textbf{RED} : $\mathit{RED}(indice, valeur)$

    \bigskip

    \textbf{Cache générique :} pas de support particulier pour les opérations de réduction

    \bigskip

    \arrow[\bfseries\color{Red}][0][Red]{Cache de réduction} (ex. : \cref{ref:PHI} \cref{ref:SortCache})

    \point[][4][Red]{Supporte une \textbf{nouvelle instruction RED} : $\mathit{RED}(adresse, valeur)$}

    \point[][4][Red]{L'opération de réduction est \textbf{traitée dans le cache}}

    \arrow[][8]{Possède une \textbf{unité de calcul}}

    \vspace*{0.5cm}
    \dref{ref:PHI}{A. Mukkara, N. Beckmann, and D. Sanchez. 2019. PHI: Architectural Support for Synchronization- and Bandwidth-Efficient Commutative Scatter Updates. In Proceedings of the 52nd Annual IEEE/ACM International Symposium on Microarchitecture (MICRO-52). Association for Computing Machinery, New York, NY, USA, 1009–1022.}
    \vspace{3pt}
    \dref{ref:SortCache}{S. Srikanth, A. Jain, T. M. Conte, E. P. Debenedictis, and J. Cook. 2021. SortCache: Intelligent Cache Management for Accelerating Sparse Data Workloads. ACM Trans. Archit. Code Optim.18, 4, Article 56 (sep 2021), 24 pages.}
    
\end{frame}
